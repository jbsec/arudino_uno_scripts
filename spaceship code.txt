int switchState = 0;

void setup() {
  pinMode(3, OUTPUT);
  pinMode(4, OUTPUT);
  pinMode(5, OUTPUT);
  pinMode(2, INPUT);
}


void loop(){
  switchState = digitalRead(2);
  //comment for some reason?

if (switchState == LOW) {
  // the button isnt pressed

  digitalWrite(3, HIGH); // green led
  digitalWrite(4, LOW); // red led
  digitalWrite(5, LOW); // red led
}

else { //button is pressed
  digitalWrite(3, LOW);
  digitalWrite(4, LOW);
  digitalWrite(5, HIGH);

  delay(250); // wait for 250 miliseconds or 1/4 a second
  // toggle leds
  digitalWrite(4, HIGH);
  digitalWrite(5, LOW);
  delay(250); // wait for 1/4 a second
  }
} // go back to beginning of loop
